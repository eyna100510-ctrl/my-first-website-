    <!-- Core Web Application Logic Script -->
    <script>
        // Model constants for the Sigmoidal growth curve equations
        const L = 2500;   // Carrying capacity (maximum bunch weight in grams)
        const k = 0.08;   // Growth rate coefficient multiplier
        const x0 = 70;    // Center point of maximum growth inflection (Day 70)

        let chartInstance = null;
        const totalDays = 140; // Full standard period of fruit bunch tracking

        // 1. Math Functions
        function getCumulativeWeight(day) {
            return L / (1 + Math.exp(-k * (day - x0)));
        }

        function getGrowthRateDerivative(day) {
            const exponentTerm = Math.exp(-k * (day - x0));
            const denominator = Math.pow(1 + exponentTerm, 2);
            return (L * k * exponentTerm) / denominator;
        }

        // 2. Generate Data Arrays for Chart.js
        const labels = [];
        const weightData = [];
        const rateData = [];

        for (let day = 0; day <= totalDays; day += 2) {
            labels.push(day);
            weightData.push(getCumulativeWeight(day).toFixed(1));
            rateData.push(getGrowthRateDerivative(day).toFixed(2));
        }

        // 3. Build Chart.js Graph Instance
        function renderChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');
            
            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [
                        {
                            label: 'Total Weight f(x) (grams)',
                            data: weightData,
                            borderColor: '#10b981', // Emerald-500
                            backgroundColor: 'rgba(16, 185, 129, 0.05)',
                            borderWidth: 3,
                            yAxisID: 'yWeight',
                            fill: true,
                            tension: 0.2,
                            pointRadius: 0,
                            pointHoverRadius: 6
                        },
                        {
                            label: "Growth Velocity f'(x) (g/day)",
                            data: rateData,
                            borderColor: '#6366f1', // Indigo-500
                            borderDash: [5, 5],
                            borderWidth: 2,
                            yAxisID: 'yRate',
                            tension: 0.2,
                            pointRadius: 0,
                            pointHoverRadius: 6
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    onClick: handleChartClick,
                    plugins: {
                        legend: { position: 'top', labels: { boxWidth: 12, font: { size: 11 } } },
                        tooltip: { mode: 'index', intersect: false }
                    },
                    animation: {
                        duration: 2500,
                        easing: 'easeOutQuart'
                    },
                    scales: {
                        x: {
                            title: { display: true, text: 'Time Sequence (Days Post-Anthesis)', font: { weight: 'bold' } },
                            grid: { color: 'rgba(0,0,0,0.03)' }
                        },
                        yWeight: {
                            type: 'linear',
                            position: 'left',
                            title: { display: true, text: 'Total Bunch Weight (g)', font: { weight: 'bold', color: '#10b981' } },
                            min: 0,
                            max: 2700
                        },
                        yRate: {
                            type: 'linear',
                            position: 'right',
                            title: { display: true, text: 'Growth Velocity Rate (g/day)', font: { weight: 'bold', color: '#6366f1' } },
                            min: 0,
                            max: 60,
                            grid: { drawOnChartArea: false } // Avoid grid lines overlay clashing
                        }
                    }
                }
            });
        }

        // 4. Handle Canvas Clicking Actions
        function handleChartClick(event, elements) {
            // Use Chart.js helper extensions to get actual math scale coordinates from pixels
            const points = chartInstance.getElementsAtEventForMode(event, 'index', { intersect: false }, true);
            
            if (points.length > 0) {
                const dataIndex = points[0].index;
                const activeDay = labels[dataIndex];
                
                calculateAndShowMetrics(activeDay);
            }
        }

        // 5. Derivative Math Interpretation Logic
        function calculateAndShowMetrics(day) {
            const currentWeight = getCumulativeWeight(day);
            const currentRate = getGrowthRateDerivative(day);
            
            // Update UI numeric readouts
            document.getElementById('resultDayBadge').innerHTML = `<i class="fa-regular fa-clock"></i> Day Evaluation Metrics: ${day}`;
            document.getElementById('outWeight').innerText = Math.round(currentWeight).toLocaleString();
            document.getElementById('outRate').innerText = currentRate.toFixed(2);

            // Generate customized localized layman text based on numerical value ranges
            let descriptionText = "";
            if (day < 45) {
                descriptionText = `The oil palm bunch is in its lag phase. Cell division is occurring internally, but overall weight gain is slow at ${currentRate.toFixed(1)} grams/day. It is too early to expect raw agro-product yield optimization during this phase.`;
            } else if (day >= 45 && day <= 64) {
                descriptionText = `The fruit bunch has entered the exponential acceleration window. The absolute derivative is surging upward. The metabolic demand of the tree is climbing rapidly as it pumps carbohydrates into the fruit matrix.`;
            } else if (day >= 65 && day <= 75) {
                descriptionText = `CRITICAL INFLECTION POINT reached! Notice how the derivative $f'(x)$ peaks right here near Day 70 at roughly 50g/day. This represents the absolute highest velocity of biomaterial expansion. Downstream food processing properties (oil synthesis inside the mesocarp) are accelerating at peak speed right now.`;
            } else if (day > 75 && day <= 105) {
                descriptionText = `The growth velocity is decelerating ($f'(x)$ drops down), though the total bunch weight continues to increase towards maturity. The physiological process shifts from expanding size to oil accumulation and refinement inside the kernel.`;
            } else {
                descriptionText = `The system has reached its asymptote plateau (Carrying capacity of ~2,500g). The rate of change is crawling near zero (${currentRate.toFixed(2)} g/day). Growth has ceased; the crop is physiologically ripe and ready for harvesting and milling extraction.`;
            }

            document.getElementById('outExplain').innerText = descriptionText;

            // Highlight and show the block container element
            const resultCard = document.getElementById('resultCard');
            resultCard.classList.remove('hidden');
            setTimeout(() => {
                resultCard.classList.remove('scale-95', 'opacity-0');
                resultCard.classList.add('scale-100', 'opacity-100');
            }, 10);

            // Turn off the flashing placeholder callout once user successfully interacts
            document.getElementById('clickPrompt').classList.remove('animate-pulse');
            document.getElementById('clickPrompt').classList.add('bg-slate-100', 'text-slate-600');
            document.getElementById('clickPrompt').innerHTML = `<i class="fa-solid fa-circle-check text-emerald-600"></i> Click any other location along the curve to run comparative analyses.`;
        }

        // Action helper to re-trigger chart loading effects
        function restartAnimation() {
            if(chartInstance) {
                chartInstance.destroy();
            }
            renderChart();
        }

        // Initial launch setup
        window.onload = function() {
            renderChart();
        };
    </script>
